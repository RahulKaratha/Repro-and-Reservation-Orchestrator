 users | CREATE TABLE `users` (
  `ID` int NOT NULL AUTO_INCREMENT,
  `First_Name` varchar(10) NOT NULL,
  `Last_Name` varchar(10) DEFAULT NULL,
  `Email` varchar(100) NOT NULL,
  `Password` varchar(255) NOT NULL,
  `Role` enum('Engineer','Manager') NOT NULL,
  PRIMARY KEY (`ID`),
  UNIQUE KEY `Email` (`Email`),
  CONSTRAINT `users_chk_1` CHECK (regexp_like(`Email`,_utf8mb4'^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\\.[A-Za-z]{2,}$')),
  CONSTRAINT `users_chk_2` CHECK ((regexp_like(`Password`,_utf8mb4'[A-Za-z]') and regexp_like(`Password`,_utf8mb4'[0-9]') and regexp_like(`Password`,_utf8mb4'[@$!%*?&]') and (length(`Password`) >= 8)))
)

 workgroup_assignment_schema | CREATE TABLE `workgroup_assignment_schema` (
  `ID` int NOT NULL AUTO_INCREMENT,
  `Workgroup_ID` int NOT NULL,
  `Employee_ID` int NOT NULL,
  PRIMARY KEY (`ID`),
  KEY `fk_workgroup` (`Workgroup_ID`),
  KEY `fk_employee` (`Employee_ID`),
  CONSTRAINT `fk_employee` FOREIGN KEY (`Employee_ID`) REFERENCES `users` (`ID`) ON DELETE CASCADE ON UPDATE CASCADE,
  CONSTRAINT `fk_workgroup` FOREIGN KEY (`Workgroup_ID`) REFERENCES `workgroup_schema` (`ID`) ON DELETE CASCADE ON UPDATE CASCADE
)
